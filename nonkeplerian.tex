\begin{wordonframe}{Momento inerzia e $J_2$}

\begin{align*}
&I_x=I_y=A=\int(x'^2+z'^2)\rho\,d^3r'=\int(y'^2+z'^2)\rho\,d^3r'\\
&I_z=C=\int(x'^2+y'^2)\rho\,d^3r'\\
&J_2=\frac{C-A}{MR^2}\, GmM=k^2\mu\\
\end{align*}
